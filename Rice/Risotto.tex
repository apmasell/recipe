\begin{recipe}{Risotto}{Nonna}{}

\begin{ingredients}
\item 1~onion, finely chopped
\item 2~portabello \theme{mushrooms}, sliced~(optional)
\item \Tp{1}$+$\tp{1} butter
\item \C{1} Arborio \theme{rice}
\item \C{1} white wine
\item chicken stock
\item \htheme{Parmesan}{cheese}, grated
\item salt
\end{ingredients}

\begin{directions}
\item Saut\'e onions and mushrooms in \Tp{1} butter.
\item Add rice and stir.
\item Add white wine.
\item Add stock in \C{\half} increments, waiting for complete absorption. Cook until rice is tender.
\item Mix in remaining butter and cheese.
\item Salt to taste.
\end{directions}

\hint{For the stock, it is easiest to dissolve 1~bullion cube in \C{2} hot water. Once this is used, use plain hot water until rice is cooked.}
\end{recipe}
