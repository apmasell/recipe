\begin{recipe}{Shrimp Macaroni Salad}{Mama Leite}{}

\begin{ingredients}
\item \lbs{\half} fresh \theme{shrimp}, boiled, shelled, and chopped
\item \lbs{1} rigatoni or other spiral pasta
\item 1~small \theme{carrot}, shredded
\item red onion, chopped~(to taste, but be liberal)
\item juice of \half{}~lemon or lime
\item Miracle Whip
\item salt
\item pepper
\end{ingredients}

\begin{directions}
\item Cook pasta in water used to boil shrimp until al dente.
\item Drain pasta.
\item Combine all ingredients except Miracle Whip.
\item Add Miracle Whip until desired creaminess is reached.
\end{directions}

\hint{Chopped green peppers can be added in addition to, or in place of, carrot.}
\end{recipe}
