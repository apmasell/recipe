\begin{recipe}{Spinach-Cauliflower Toss}{}{}

\begin{ingredients}
\item \C{\half} pine nuts or slivered \shtheme{almond}{nuts}s
\item \half~bunch of \theme{spinach}, torn into bits
\item \half~medium \theme{cauliflower}, cut into florets and sliced \inch{1\quarter}
\item 1~large \theme{avocado}, cubed
\item lemon juice
\item \Tp{6} oil
\item \Tp{3} wine vinegar
\item clove of garlic, minced
\item \tp{\half} salt
\item \tp{\half} dry mustard
\item \tp{\half} dried basil
\item \tp{\quarter} pepper
\item dash of nutmeg
\end{ingredients}

\begin{directions}
\item Toast nuts at \tF{350} for 8~minutes.
\item Dip avocado in lemon juice to coat.
\item Combine oil, vinegar, and seasonings.
\item Combine all ingredients and toss gently.
\end{directions}

\end{recipe}
