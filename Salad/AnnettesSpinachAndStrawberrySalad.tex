\begin{recipe}{Annette's Spinach and Strawberry Salad}{}{}

\begin{ingredients}
\item 1~spring onion
\item \theme{strawberries}, sliced
\item \theme{spinach}
\end{ingredients}

Dressing
\begin{ingredients}
\item \C{\half} olive oil
\item \Tp{2} white vinegar
\item \Tp{2} red wine vinegar
\item \Tp{2} poppy seeds
\item \tp{1} Worcestershire sauce
\item \C{\quarter} sugar
\item dash of paprika
\end{ingredients}

\begin{directions}
\item Whisk dressing ingredients together.
\item Combine all ingredients and toss.
\end{directions}

\end{recipe}
