\begin{recipe}{Tortellini Salad}{}{}

\begin{ingredients}
\item \lbs{1} green tortellini, fresh or frozen
\item 4~tomatoes, seeded and chopped
\item \lbs{\half} \theme{fontina cheese}, diced or grated
\item \half~package frozen \theme{spinach}, thawed and drained
\item \tp{\half} salt
\item pepper
\item \C{1} oil
\item \C{\half} white vinegar
\item \Tp{2} Dijon mustard
\item \Tp{2} parsley, chopped
\item \Tp{\half} garlic, chopped
\end{ingredients}

\begin{directions}
\item Cook, drain, and rinse pasta.
\item Combine oil, vinegar, mustard, parsley, and garlic in a blender for 1~minute.
\item Toss everything together.
\item Season to taste.
\item Chill.
\end{directions}

\end{recipe}
