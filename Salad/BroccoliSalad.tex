\begin{recipe}{Broccoli Salad}{Lorraine Teicht}{}

\begin{ingredients}
\item 1~\theme{broccoli}, cut into bite-sized pieces
\item \half~red onion, diced
\item \lbs{\half} \theme{bacon}, fried and chopped
\item \C{\half} \htheme{cheddar}{cheese}, shredded
\item \C{\half} mayonnaise
\item \C{\half} sugar
\item \Tp{2} white vinegar
\end{ingredients}

\begin{directions}
\item Whisk together mayonnaise, sugar, and vinegar.
\item Combine all ingredients and toss.
\item Chill.
\end{directions}

\end{recipe}
