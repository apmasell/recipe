\begin{recipe}{Bechamel}{}{}

\begin{ingredients}
\item \Tp{1} butter
\item \Tp{1} flour
\item \C{1} milk
\item salt
\item pepper
\end{ingredients}

\begin{directions}
\item Melt butter in a saucepan.
\item Blend in flour.
\item Add milk and seasonings.
\item Simmer, stirring constantly, until thickened.
\end{directions}

\end{recipe}
