\begin{recipe}{Chocolate Sauce}{Alton Brown}{\C{1\half}}

\begin{ingredients}
\item \C{1} sugar
\item \C{\half} water
\item \C{\half} Dutch processed cocoa, sifted
\item \tp{\eighth} salt
\end{ingredients}

\begin{directions}
\item Combine sugar, salt, and water in a small pot.
\item Bring to a boil.
\item Cover and cook for 2~minutes.
\item Slowly add cocoa while whisking vigorously.
\item When the cocoa melts and combines, remove from heat.
 \end{directions}

\hint{Dutch processed cocoa, or alkalized cocoa, has been treated with base to reduce the acidity. It is much darker than natural cocoa, the kind usually found in grocery stores.}
\end{recipe}
