\begin{recipe}{Battered Zucchini}{}{}

\begin{ingredients}
\item \theme{zucchini}, coined or julienned
\item egg, beaten
\item all-purpose flour
\item olive oil
\item salt
\end{ingredients}

\begin{directions}
\item Salt zucchini.
\item Coat with flour, then egg.
\item Fry in oil.
\end{directions}

\end{recipe}
