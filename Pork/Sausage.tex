\begin{recipe}{Pork Sausage}{Nonna}{}

\begin{ingredients}
\item lean ground \theme{pork}
\item salt
\item pëpparulë
\item coriander
\item black pepper
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item Keep in fridge for at least 24~hours.
\item Use to fill sausage casings.
\end{directions}

\hint{This can be used directly as the meat base for \seerecipe{Bread:Calzone}.}
\end{recipe}
