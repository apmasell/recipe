\begin{recipe}{Irish Soda Bread}{}{1~loaf}

\begin{ingredients}
\item \gr{175} all-purpose flour
\item \gr{175} whole wheat flour
\item \gr{50} rolled oats
\item \tp{1} salt
\item \tp{1} baking soda
\item \gr{50} butter, melted
\item \mL{185} buttermilk
\end{ingredients}

\begin{directions}
\item Prehead oven to \tF{400}.
\item Combine dry ingredients.
\item Add wet ingredients to dry.
\item Mix, then knead for 1 to 2~minutes.
\item Shape into a round.
\item Deeply cut a cross in the top of the round.
\item Bake for 20 to 25~minutes.
\end{directions}

\end{recipe}
