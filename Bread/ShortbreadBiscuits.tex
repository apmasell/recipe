\begin{recipe}{Shortbread Biscuits \UNTESTED}{}{4 biscuits}

\begin{ingredients}
\item \C{2} all-purpose flour
\item \tp{\half} salt
\item \Tp{2} baking powder
\item \Tp{2} sugar
\item \Tp{3} butter
\item \C{\half} milk
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{400}.
\item Combine dry ingredients.
\item Blend in butter.
\item Stir in milk.
\item Roll out to to \cm{2} thick and cut into quarters.
\item Bake for 15~minutes.
\end{directions}
\end{recipe}
