\begin{recipe}{Sweet Potato Rolls}{Floyd M}{20~rolls}

\begin{ingredients}
\item \gr{185} \theme{sweet potato}, riced
\item \C{1} milk
\item \C{\half} sugar
\item \gr{9} compressed yeast
\item \tp{2} salt
\item \tp{\half} cinnamon
\item \tp{\eighth} nutmeg
\item \gr{200} all-purpose flour
\end{ingredients}

\begin{directions}
\item Combine sweet potato, milk, and sugar.
\item Beat until sugar dissolves.
\item Add yeast, salt, cinnamon, and nutmeg.
\item Stir until yeast dissolves.
\item Add flour.
\item Knead until smooth and elastic.
\item Let rise for 45~minutes.
\item Divide dough and shape into rolls.
\item Let proof for 1~hour.
\item Preheat oven to \tF{375}.
\item Bake until golden on top or internal temperature reaches \tF{207}.
\end{directions}

\hint{These are quite sweet. Too sweet for dinner rolls.}
\end{recipe}
