\begin{recipe}{Stuffed Squid\UNTESTED}{}{}

\begin{ingredients}
\item \lbs{2} fresh \theme{squid}
\item \C{2\half} bread crumbs
\item \tp{2} parsley
\item \Tp{2} \htheme{Parmesan}{cheese}, grated
\item \Tp{2} oil
\item 1~medium red onion, chopped
\item salt
\item pepper
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{375}.
\item Clean squid.
\item Combine remaining ingredients and stuff squid.
\item Pour \C{\half} oil and squid in a shallow baking dish.
\item Bake uncovered for 35~minutes.
\end{directions}

\end{recipe}
