\begin{recipe}{Pimento Clams}{Fernando Leite}{}

\begin{ingredients}
\item \theme{clams} or \theme{mussels}
\item olive oil
\item pimento paste
\item garlic
\item parsley
\item white wine
\item \theme{beer}
\item salt
\end{ingredients}

\begin{directions}
\item Saut√© pimento pasted and garlic in olive oil.
\item Add clams and cook slightly.
\item Add white wine.
\item Add parsley and salt.
\item Add beer.
\item Cook until clams are open.
\end{directions}

\hint{Be careful with the salt since the paste and beer both have salt.}
\end{recipe}
