\begin{recipe}{Vanilla Velvet Cookies}{}{}

\begin{ingredients}
\item \C{1} unsalted butter
\item \oz{3} \htheme{cream}{cheese}
\item \C{1} sugar
\item 1~egg yolk
\item \tp{1} vanilla
\item \C{2\half} flour
\item salt
\end{ingredients}

\begin{directions}
\item Cream together butter and cream cheese.
\item Gradually add sugar, egg yolk and vanilla.
\item Gradually stir in flour and salt.
\item Work dough into a ball.
\item Divide dough into two thick discs.
\item Refrigerate discs wrapped in plastic for 1~hour.
\item Preheat oven to \tF{350}.
\item Use a flour rolling pin to roll chilled dough to \cm{\half} thickness.
\item Cut into shapes and place on a cookie sheet.
\item Bake until cookies are just turning golden at edges. Approximately 12~minutes. Do not over bake.
\end{directions}
\method{Creaming}
\end{recipe}
