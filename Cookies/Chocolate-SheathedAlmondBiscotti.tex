\begin{recipe}{Chocolate-Sheathed Almond Biscotti}{}{4\half\ dozen}

\begin{ingredients}
\item 3~eggs
\item \tp{1} vanilla
\item \tp{\quarter} almond extract
\item \C{2\quarter} all-purpose flour
\item \C{\threequarter}$+$\Tp{2} sugar
\item \tp{1} baking soda
\item \tp{\half} salt
\item \tp{2} cinnamon
\item \C{\threequarter} \shtheme{almond}{nuts}s, toasted and chopped into halves or thirds
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{325}.
\item Grease and flour a baking sheet.
\item In a small bowl, using a whisk, beat together the eggs, vanilla and almond extract until blended.
\item In a large bowl, stir together the flour, sugar, baking soda, salt, and cinnamon.
\item Add the egg mixture to the flour mixture and mix until blended.
\item Stir in the nuts.
\item Divide the dough in half.
\item One at a time, place the 2~dough portions on the prepared baking sheet and form each into a log \inch{\half} high, \inch{1 \half} wide, and \inch{14} long.
\item Bake the logs until set and golden brown. Approximately 25~minutes.
\item Transfer to a rack and let cool on the baking sheet for 6 to 8~minutes.
\item Reduce the oven temperature to \tF{300}.
\item Transfer the logs to a cutting board.
\item Using a serrated knife, slice diagonally \cm{\half} thick.
\item Lay the slices flat on the baking sheet and return to the oven for 15~minutes longer, turning them once, to dry slightly.
\item Transfer to racks to cool.
\end{directions}

\subrecipe{Chocolate Glaze}

\begin{ingredients}
\item \oz{6} bitter-sweet \theme{chocolate}
\item \tp{\half} vegetable shortening
\end{ingredients}

\begin{directions}
\item In the top pan of a double boiler, combine the chocolate and shortening.
\item Place over hot water in the lower pan and heat until melted, then stir until smooth.
\item Using an icing spatula, spread the chocolate over the top surface of the cookies.
\item Let cool until set.
\end{directions}

\hint{Store the cookies between the sheets of waxed paper in an airtight container for up to 1~week.}
\end{recipe}
