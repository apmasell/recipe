\begin{recipe}{Biscotti Neri}{Zia Graziella}{}

\begin{ingredients}
\item \C{6} all-purpose flour
\item \C{1} sugar
\item \lbs{\half} melted shortening
\item \oz{16} tepid milk
\item \Tp{3} \theme{cocoa}
\item \tp{2} cinnamon
\item \tp{1} cloves
\item \Tp{6} baking powder
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item Mix dough by hand.
\item With oil-covered hands, roll into ``snakes''.
\item Cut diagonally.
\item Grease cookie sheets or line with foil.
\item Bake at \tF{350} for 10 to 15~minutes---no more.
\end{directions}

\subrecipe{Icing}

\begin{ingredients}
\item confectioner's sugar
\item water
\item lemon juice
\end{ingredients}

\begin{directions}
\item Once cookies are cool (about 1~hour), mix icing in a bowl. It should have the consistency of corn syrup.
\item Add cookies and toss gently to coat.
\item Let stand until it begins to harden.
\item Lay cookies out and let dry thoroughly before storing.
\end{directions}

\hint{These cookies freeze well if layered between waxed paper.}
\hint{Because of their appearance when unbaked, Jessica called these ``poo-poo'' cookies.}
\end{recipe}
