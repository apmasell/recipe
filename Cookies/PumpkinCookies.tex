\begin{recipe}{Pumpkin Cookies}{}{}

\begin{ingredients}
\item \C{\quarter} butter, softened
\item \C{\half} light brown sugar
\item \C{\half} \theme{pumpkin} purée
\item 1~egg
\item \C{1} flour
\item \C{\half} \theme{raisins}
\item \C{\half} \shtheme{almond}{nuts}s or \shtheme{walnut}{nuts}s, chopped
\item \tp{2} baking powder
\item \tp{1} cinnamon
\item \tp{\quarter} \theme{ginger}
\item \tp{\quarter} nutmeg
\item \tp{\quarter} salt
\item \C{\quarter} candied orange peel, diced~(optional)
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Sift flour, baking powder, cinnamon, ginger, nutmeg and salt together.
\item In a large mixing bowl, beat butter and sugar until fluffy.
\item Beat in egg and pumpkin.
\item Stir in flour mixture.
\item Add in nuts, orange peel and raisins, mixing well.
\item Drop spoonfuls of dough onto a cookie sheet.
\item Bake 20~minutes.
\end{directions}
\end{recipe}
