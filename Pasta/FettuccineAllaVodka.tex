\begin{recipe}{Fettuccine alla Vodka}{Centro}{}

\begin{ingredients}
\item \lbs{3} fresh fettuccine or linguine
\item \Tp{2} olive oil
\item \Tp{2} butter
\item clove of garlic, crushed
\item \Tp{1} onions, finely chopped
\item 4~medium tomatoes, diced
\item \oz{8} smoked \shtheme{salmon}{fish}, chopped
\item \Tp{4} \shtheme{vodka}{alcohol}
\item \C{1} heavy \theme{cream}
\item pepper
\item \C{\quarter} \htheme{Parmesan}{cheese}
\end{ingredients}

\begin{directions}
\item Heat butter and oil in a pan.
\item Add garlic and onions.
\item Saut\'e tomatoes for 5~minutes.
\item Add salmon, vodka, and cream.
\item Cook 10~minutes.
\item Cook pasta and drain.
\item Toss with sauce.
\item Add pepper to taste and cheese.
\end{directions}

\end{recipe}
