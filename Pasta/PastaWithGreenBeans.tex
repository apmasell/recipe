\begin{recipe}{Pasta with Green Beans}{}{}

\begin{ingredients}
\item buccatini or long pasta
\item salt
\item green or wax \theme{beans}
\item \seerecipe{PastaSauce:FreshTomatoSauce} or \seerecipe{PastaSauce:PlainTomatoSauce}
\end{ingredients}

\begin{directions}
\item Wash beans and remove tips.
\item Bring a pot of water to a boil.
\item Salt the water.
\item Add pasta and beans.
\item Cook pasta for length of time specified on package.
\item Drain.
\item Toss in sauce.
\end{directions}

\end{recipe}
