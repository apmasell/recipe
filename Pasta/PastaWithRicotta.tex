\begin{recipe}{Pasta with Ricotta}{Nonna}{}

\begin{ingredients}
\item \htheme{ricotta}{cheese}
\item pasta, rúcchëlë recommended\seerecipe{Pasta:Rucchele}
\item parsley~(optional)
\item \htheme{Parmesan}{cheese}~(optional)
\end{ingredients}

\begin{directions}
\item Place ricotta in serving bowl.
\item Bring salted water to a boil.
\item Cook pasta until tender.
\item Thin ricotta with pasta water.
\item Drain pasta.
\item Toss pasta with ricotta.
\item Add parsley and Parmesan and toss.
\end{directions}

\hint{Nonna dislikes the parsley and Parmesan, but ``some people'' add it.}
\end{recipe}
