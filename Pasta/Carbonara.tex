\begin{recipe}{Carbonara}{Nonna}{4 to 6~servings}

\begin{ingredients}
\item \gr{450} spaghetti
\item 6~slices of \theme{bacon}, chopped
\item 3~eggs
\item \C{\half} water
\item \C{\twothird} \htheme{Parmesan}{cheese} and \htheme{Romano}{cheese}, grated and mixed
\item \C{\half} white wine
\item salt
\item pepper
\end{ingredients}

\begin{directions}
\item Cook pasta until done.
\item Meanwhile, in a skillet, fry bacon.
\item When bacon is crisp, add white wine.
\item Reduce heat and let wine evaporate.
\item Beat together eggs, water, and cheese, adding salt and pepper to taste.
\item Drain pasta, reserving at least \C{1} pasta water, and return pasta to pot. Do not heat pasta.
\item Add bacon mixture, egg mixture, and pasta water as needed.
\item Stir quickly until egg cooks.
\end{directions}

\hint{Add reserved pasta water to reach desired creaminess, adding extra, as pasta will absorb a considerable amount of water.}
\end{recipe}
