\begin{recipe}{Butter Tarts}{Diane Leite}{}

\begin{ingredients}
\item tart shells
\item 3~eggs
\item \C{1} brown sugar
\item \C{\half} corn~syrup
\item \C{\quarter} butter, softened
\item \tp{\half} salt
\item \tp{1} vanilla
\item \C{1} or more \theme{nuts} and/or \theme{raisins}
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Bake shells for 10~minutes
\item Combine remaining ingredients.
\item Pour mixture into shells.
\item Bake for 20~minutes.
\end{directions}

\end{recipe}
