\begin{recipe}{Open Apple Pie}{Jim Leising}{1 pie}

\begin{ingredients}
\item pie shell
\item \C{\half} brown sugar
\item \C{\half} sugar
\item \Tp{1} lemon rind, coarsely grated
\item \tp{\half} nutmeg
\item \tp{\half} cinnamon
\item pinch of salt
\item \C{6} \theme{apple}s, peeled and sliced
\item \C{\third} whipping \theme{cream}
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{400}.
\item Stir together sugars, lemon rind, nutmeg, cinnamon, and salt.
\item Arrange apples in pie shell.
\item Sprinkle sugar over apples.
\item Drizzle cream over apples.
\item Bake for 35 to 45~minutes.
\end{directions}

\end{recipe}
