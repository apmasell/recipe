\begin{recipe}{Lemon Granita}{}{\mL{500}}

\begin{ingredients}
\item \C{\twothird} lemon juice, fresh
\item \C{1\third} water
\item \C{\half} sugar
\item \Tp{2} lemon zest
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item Stir until sugar is dissolved.
\item Pour in a shaker or a flat metal pan.
\item Place in the freezer.
\item Every hour, shake or stir until frozen.
\end{directions}

\hint{If granita freezes solid, use a fork to break it into grains. This is the traditional method.}
\end{recipe}
