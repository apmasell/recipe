\begin{recipe}{Vanilla Pecans}{Canadian Living}{}

\begin{ingredients}
\item \C{4} \shtheme{pecan}{nuts}s or \shtheme{walnut}{nuts}s
\item \C{\quarter} sugar
\item \Tp{1} oil
\item \Tp{1} vanilla
\item \tp{1} cinnamon
\item \tp{\half} nutmeg
\item \tp{\quarter} salt
\item cayenne pepper
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{300}.
\item In a large pot of boiling water, blanch nuts for 1~minute.
\item Drain nuts.
\item Combine sugar, oil, vanilla.
\item Toss with pecans.
\item Spread on baking sheet and bake for 20~minutes.
\item Stir and bake for 20~minutes.
\item Combine cinnamon, nutmeg, salt and a pinch of pepper.
\item Toss with pecans.
\item Bake for 5~minutes.
\item Let cool.
\end{directions}

\hint{Store in at room-temperature for up to 1~week, or frozen for 1~month.}
\end{recipe}
