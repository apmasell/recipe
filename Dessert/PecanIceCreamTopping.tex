\begin{recipe}{Pecan Ice Cream Topping}{Daiva Paulionis}{}

\begin{ingredients}
\item \C{\half} \shtheme{pecan}{nuts}, chopped
\item \Tp{1-1\half} sugar
\item \tp{\quarter} salt
\item \Tp{1} butter
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item Heat until pecans begin to toast.
\item Sprinkle over vanilla ice cream with caramel sauce.
\end{directions}

\end{recipe}
