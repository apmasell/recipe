\begin{recipe}{Orange and Rum Pizzelles}{}{50~pizzelles}

\begin{ingredients}
\item 3~eggs
\item \C{1} sugar
\item \tp{2} dark \theme{rum}
\item \C{\half} melted butter, cooled
\item \C{2} all-purpose flour
\item \tp{2} baking powder
\item \tp{2} grated orange peel
\end{ingredients}

\begin{directions}
\item Beat eggs and sugar.
\item Slowly add melted butter.
\item Add rum and orange peel.
\item Gradually add flour.
\item Drop on to pizzelle maker \tp{1} at a time.
\end{directions}

\end{recipe}
