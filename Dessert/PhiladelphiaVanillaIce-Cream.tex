\begin{recipe}{Philadelphia Vanilla Ice-Cream}{}{\mL{500}}

\begin{ingredients}
\item \mL{500} half \& half \theme{cream}
\item \C{\threequarter} sugar
\item \tp{1} vanilla
\item pinch salt
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item If sugar does not dissolve, warm mixture until sugar dissolves, then chill.
\item Churn in an ice-cream maker.
\item Pour in a container and allow to freeze hard.
\end{directions}

\end{recipe}
