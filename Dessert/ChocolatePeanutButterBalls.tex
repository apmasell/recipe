\begin{recipe}{Chocolate Peanut Butter Balls}{}{}

\begin{ingredients}
\item \C{1} \theme{peanut butter}
\item \C{1} confectioner's sugar
\item \C{\half} \theme{nuts}, chopped
\item \C{\half} puffed rice \theme{cereal}
\item \oz{4} semi-sweet \theme{chocolate}
\item \Tp{1} butter
\end{ingredients}

\begin{directions}
\item Combine all ingredients except chocolate and butter.
\item Chill for 10 to 15~minutes.
\item Roll into balls.
\item Melt chocolate and butter.
\item Dip balls into melted chocolate and chill.
\end{directions}
\end{recipe}
