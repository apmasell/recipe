\begin{recipe}{Pizzelle\FIXME}{Zia Franca}{}

\begin{ingredients}
\item 10~eggs less 4~whites
\item \C{1} less \Tp{1} oil
\item \C{1} less \Tp{1} sugar
\item lemon zest
\item \oz{1} liqueur
\item cake-and-pastry flour
\end{ingredients}

\begin{directions}
\item Beat liquid ingredients.
\item Add enough flour to make consistency of pancake batter.
\end{directions}

\hint{The ``exact'' amount is \Tp{1\half} each sugar and oil per egg, but using serving spoons, not calibrated ones.}
\hint{Substitute cocoa for \Tp{3} flour to make chocolate pizzelle.}
\end{recipe}
