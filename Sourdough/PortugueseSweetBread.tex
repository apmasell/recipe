\begin{recipe}{Portuguese Sweet Bread}{}{}

\begin{ingredients}
\item \gr{145} starter
\item zest of a lemon
\item zest of an orange
\item \tp{1} salt
\item \gr{100} sugar
\item \tp{1} vanilla
\item 2~eggs
\item \gr{190} juice of a lemon + juice of an orange + water
\item \gr{40} milk powder
\item \gr{30} butter
\item \gr{30} shortening
\item \gr{5} baking soda
\item \gr{470} flour
\item egg wash
\end{ingredients}

\begin{directions}
\item Combine juices, water, milk powder, butter, shortening, and baking soda in a sauce pan.
\item Scald, stirring frequently to control foaming.
\item Let cool.
\item Combine above mixture with all ingredients except flour and mix thoroughly.
\item Add flour to make a firm dough and knead until dough passes the window pane test.
\item Allow to rise until doubled in bulk.
\item Turn, round, and place in a greased round cake pan. (Ideally, the pan should have a lip.)
\item Allow to rise until doubled in bulk, just before dough overflows the pan.
\item Preheat oven to \tF{350}.
\item Brush with egg wash.
\item Bake for 40~minutes, or until an internal temperature of \tF{207} is reached.
\end{directions}

\hint{The total water-type liquid content must be \gr{190}. Since the juice from a lemon and orange varies, extract all the juice, then add water so that the final weight of the liquid is correct.}
\hint{This recipe is derived from Peter Reinhart's The Bread Baker's Apprentice.}
\hint{Scalding is necessary to neutralise proteases in the juice and milk. Otherwise, the texture will be crumbly.}
\hint{Baking soda is necessary to reduce the acidity of the juice. Otherwise, the dough will be slow to rise and the crumb will not have a soft, uniform texture.}
\end{recipe}
