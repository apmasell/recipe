\begin{recipe}{Pan de cristal}{Joy Ride Coffee}{}

\begin{ingredients}
\item \gr{400} flour
\item \gr{420} water
\item \gr{11} salt
\item \gr{10} olive oil
\item \gr{100} sourdough starter
\end{ingredients}

\begin{directions}
\item Combine salt and \gr{70} water and chill.
\item Combine starter, flour, and \gr{350} water.
\item In a stand mixer, knead until smooth using a paddle.
\item Slowly add chilled brine and olive oil.
\item Let bulk ferment, turning every hour.
\item Place in a sealed container and chill for 24~hours.
\item Turnout onto a sheet of parchment paper.
\item Let rise for 1~hour.
\item Preheat oven to \tF{425}.
\item Bake for 25~minutes.
\end{directions}

\hint{This is prepared almost like a brioche dough, but adding water instead of fat.}
\end{recipe}
