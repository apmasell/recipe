\begin{recipe}{Waffles}{}{10}

\begin{ingredients}
\item \gr{150} starter
\item 2~eggs
\item \gr{340} and \Tp{1} water
\item \gr{340} all-purpose flour
\item \gr{8} salt
\item \gr{40} sugar
\item \gr{100} oil
\item \tp{1} baking soda
\end{ingredients}

\begin{directions}
\item Combine all ingredients except \Tp{1} water and baking soda.
\item Let rise overnight.
\item Dissolve baking soda in water and mix into batter.
\item Cook in a waffle iron.
\end{directions}

\end{recipe}
