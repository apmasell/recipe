\begin{recipe}{Gazpacho}{the Steffler Family}{}

\begin{ingredients}
\item \C{3} salsa
\item \C{1} green pepper
\item \C{1} cucumber
\item \C{\quarter} green onion
\item \C{} red pepper
\item \C{1} celery
\item \C{1} tomato
\item 1 clove minced garlic
\item \C{2} water
\item salt and pepper
\end{ingredients}

\begin{directions}
\item Finely chop all ingredients.
\item Combine and chill for at least 2~hours.
\item Serve cold.
\end{directions}
\end{recipe}
