\begin{recipe}{Apple Kuchen\UNTESTED}{Mary McGrath}{}

\begin{ingredients}
\item \C{\half} butter
\item \C{1} sugar
\item \Tp{1} sugar
\item 2~eggs
\item \C{1} all-purpose flour
\item \tp{1} baking powder
\item 2 to 3~\theme{apple}s, peeled and thinly sliced
\item \Tp{1} lemon or lime juice
\item \tp{\half} cinnamon
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Cream butter and \C{1} sugar.
\item Beat in eggs.
\item Mix in flour and baking powder.
\item Pour into a \inch{9} round pan.
\item Place apple slices on top of batter in circles.
\item Sprinkle with juice.
\item Mix cinnamon and \Tp{1} sugar.
\item Sprinkle sugar mixture over top.
\item Bake for 60~minutes.
\end{directions}
\method{Creaming}
\end{recipe}
