\begin{recipe}{Carrot Cake}{}{}

\begin{ingredients}
\item \C{2} flour
\item \C{2} sugar
\item \tp{2} baking soda
\item \tp{2} cinnamon
\item \tp{1} salt
\item \C{3} grated \theme{carrot}s
\item 4~eggs
\item \tp{1} vanilla
\item \C{\half} oil
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{325}.
\item Sift dry ingredients into a mixing bowl.
\item Mix in grated carrots, vanilla and eggs.
\item Beat, adding oil slowly.
\item Pour into a greased pan.
\item Bake for 35 to 45~minutes.
\end{directions}
\end{recipe}
