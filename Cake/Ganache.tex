\begin{recipe}{Ganache}{Andre Masella}{\inch{9} layer cake}

\begin{ingredients}
\item \gr{300} dark \theme{chocolate}, chopped
\item \gr{100} \theme{cream}
\end{ingredients}

\begin{directions}
\item In a double boiler, heat cream to \tC{37}.
\item Add chocolate.
\item Stir until chocolate is melted and blended. Do not let temperature exceed \tC{37}.
\item Frost cake.
\end{directions}

\hint{The heavier the cream, the better.}
\hint{This is a $3:1$ ratio which will produce a firm ganache. It can be reduced to $2:1$ for a softer ganache.}
\end{recipe}
