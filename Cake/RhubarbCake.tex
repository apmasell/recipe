\begin{recipe}{Rhubarb Cake}{}{}

\begin{ingredients}
\item \C{\half} shortening
\item \C{1\half} sugar
\item 1~egg
\item \tp{1} vanilla
\item \tp{1} baking soda
\item \C{2} flour
\item \tp{\half} salt
\item \C{1} buttermilk
\item \C{2\quarter} \theme{rhubarb}, chopped into \inch{\half} lengths
\item \Tp{1} flour
\item \C{\quarter} butter
\item \tp{2} cinnamon
\item \C{1} brown sugar
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Cream shortening and sugar.
\item Beat in egg and vanilla.
\item Combine \C{2} flour, baking soda, and salt.
\item Add alternately dry ingredients and butter milk in three additions.
\item Toss rhubarb in \Tp{1} flour.
\item Mix rhubarb into batter.
\item Spoon into a \inch{9}$\times$\inch{13} pan.
\item Combine butter, cinnamon, and brown sugar.
\item Cut until crumbly.
\item Sprinkle over batter.
\item Bake for 45~minutes.
\end{directions}

\end{recipe}
