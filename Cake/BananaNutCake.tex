\begin{recipe}{Banana Nut Cake}{Rosa Gemmel}{}

\begin{ingredients}
\item \C{2} all-purpose flour
\item \C{\half} sugar
\item \tp{3} baking powder
\item \tp{\half} baking soda
\item \tp{\half} salt
\item \C{\half} \theme{nuts}, chopped
\item 1~egg
\item \C{\threequarter} milk
\item \C{\half} oil
\item \C{1\half} \theme{banana}s, mashed
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Combine dry ingredients.
\item Separately, beat egg, milk, oil, and bananas.
\item Add liquid to dry ingredients.
\item Blend until moistened.
\item Pour into a pan.
\item Bake for 65 to 70~minutes.
\end{directions}
\method{Muffin}

\end{recipe}
