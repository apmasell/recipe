\begin{recipe}{Zucchini Cake}{}{}

\begin{ingredients}
\item 3~eggs
\item \C{1\half} sugar
\item \C{2} \theme{zucchini}, shredded
\item \C{1} oil
\item \tp{2} vanilla
\item \C{3} flour
\item \C{\half} \theme{nuts}, chopped
\item \tp{1} salt
\item \tp{1} baking soda
\item \tp{1} baking powder
\item \tp{2} cinnamon
\item \tp{\half} nutmeg~(optional)
\item \tp{\quarter} cloves~(optional)
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{325}.
\item Beat eggs until foamy.
\item Stir in sugar, zucchini, oil, and vanilla.
\item Add remaining ingredients by hand.
\item Blend well.
\item Pour into a Bundt pan.
\item Bake for 50 to 60~minutes.
\end{directions}

\end{recipe}
