\begin{recipe}{Granola Bars}{}{}

\begin{ingredients}
\item \C{\half} \theme{raisins} or \theme{dates}
\item \C{\half} shredded \theme{coconut}
\item \C{\quarter} wheat germ
\item \C{\quarter} \theme{chocolate} chips
\item \C{\quarter} \theme{nuts}, chopped
\item \tp{\half} nutmeg 
\item \tp{\half} cinnamon 
\item \tp{\half} cloves 
\item \tp{\half} \theme{ginger} 
\item \tp{1} vanilla 
\item 2~eggs
\item \C{\twothird} \theme{peanut butter}
\item \Tp{1} oil
\item \Tp{1} \theme{molasses} or \theme{honey}
\item \C{\half} water
\item \C{2} quick-cooking rolled \theme{oats}
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{300}. 
\item Line a \inch{9} baking pan with parchment paper.
\item In a large bowl blend together all ingredients except oats until well combined.
\item Add oats and mix well.
\item Press evenly into greased pan.
\item Bake for 15~minutes or until light golden brown.
\item Cool for 1~hour. 
\item Cut into bars. 
\end{directions}

\end{recipe}
