\begin{recipe}{Prime Rib Roast}{}{}

\begin{ingredients}
\item roast
\item 1~onion, halved
\item garlic powder
\item 4 to 5~potatoes
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{450}.
\item Stand roast on bone (fat-side up) in a roasting pan.
\item Dust with garlic powder.
\item Add onion halves.
\item Put roast in oven.
\item Sear for 10 to 15~minutes.
\item Reduce oven temperature to \tF{350}.
\item Add the potatoes to the pan.
\item Cook until done using a meat thermometer. Approximately \half~hour per pound.
\end{directions}

\hint{Instead of roasting in the oven, put roast on rotisserie and place pan of potatoes underneath.}
\end{recipe}
