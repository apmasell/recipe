\begin{recipe}{Meatloaf}{}{}

\begin{ingredients}
\item \lbs{1} ground \theme{beef}
\item \C{1} bread crumbs
\item 2~eggs
\item \Tp{2} \htheme{Parmesan}{cheese}, grated
\item \tp{2} salt
\item parsley, chopped
\item oil for frying
\item \seerecipe{PastaSauce:PlainTomatoSauce}
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Mix eggs, bread crumbs, cheese, parsley, salt and \C{1} sauce.
\item Knead it meat, adding sauce as needed. Mixture should be soft enough that a rolled ball will deform slightly. Not all sauce is needed.
\item Form into a loaf and place in a pan.
\item Bake for \half~hour.
\item Slice and dress each slice with more tomato sauce.
\end{directions}
\end{recipe}
