\begin{recipe}{Mussel Sauce}{Nonno}{}

\begin{ingredients}
\item \theme{mussels}
\item clove of garlic, halved
\item olive oil
\item tomato purée
\item parsley, chopped
\end{ingredients}

\begin{directions}
\item Sauté garlic in oil until golden.
\item Add tomato purée and parsley.
\item Cook 30~minutes.
\item Add mussels.
\item Cook 10~minutes.
\end{directions}

\end{recipe}
