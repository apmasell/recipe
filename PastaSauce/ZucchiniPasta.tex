\begin{recipe}{Zucchini Pasta}{Nonna}{}

\begin{ingredients}
\item 1~small \theme{zucchini} per person
\item garlic, halved
\item olive oil
\item salt
\item pepper
\item \theme{Parmesan cheese}
\end{ingredients}

\begin{directions}
\item Slice zucchini into thin coins.
\item Fry garlic in oil until golden.
\item Remove garlic.
\item Fry zucchini until golden.
\item Toss with cooked pasta and cheese
\item Season with salt and pepper.
\end{directions}

\end{recipe}
