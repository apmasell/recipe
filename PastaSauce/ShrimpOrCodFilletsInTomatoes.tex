\begin{recipe}{Shrimp or Cod Fillets in Tomatoes}{Nonno}{}

\begin{ingredients}
\item shrimp or cod fillets
\item olive oil
\item tomato pieces
\item oregano
\end{ingredients}

\begin{directions}
\item Fry garlic in oil.
\item Add tomato pieces and oregano.
\item Simmer.
\item Add fish.
\item Cook until fish is done.
\end{directions}

\hint{This recipe is also good over rice.}
\end{recipe}
