\begin{recipe}{Tortilla Chip Dip}{}{}

\begin{ingredients}
\item \oz{8} \theme{cream cheese}, softened
\item \C{\half} sour cream
\item \theme{salsa}
\item bell \theme{peppers}, chopped
\item \theme{tomatoes}, chopped
\item \theme{cheddar cheese}, shredded
\end{ingredients}

\begin{directions}
\item Blend cream cheese and sour cream.
\item Spread in a pan.
\item Add a layer of salsa.
\item Add a layer of bell peppers and tomatoes.
\item Add a layer of cheddar cheese.
\end{directions}

\end{recipe}
