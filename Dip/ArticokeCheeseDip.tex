\begin{recipe}{Artichoke Cheese Dip}{}{}

\begin{ingredients}
\item 1~can \theme{artichoke}s
\item \C{1} mayonnaise
\item \C{1\half} shredded \theme{mozzarella}
\item \C{1} \theme{Parmesan cheese}
\item \tp{1} garlic powder
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Chop artichokes in food processor.
\item Add other ingredients and blend.
\item Bake at \tF{350} for 30 to 35~minutes.
\end{directions}

\end{recipe}
