\begin{recipe}{Salmon Ball}{Maria D'Ambrosio}{}

\begin{ingredients}
\item \oz{16} tin of \shtheme{salmon}{fish}, drained
\item \Tp{1} finely chopped cooking onion
\item \tp{1} prepared horseradish
\item \tp{\quarter} salt~(optional)
\item \Tp{1} lemon juice
\item \oz{8} \htheme{cream}{cheese}, softened
\item parsley
\end{ingredients}

\begin{directions}
\item Blend all ingredients with a fork.
\item Chill and form into a ball.
\item Garnish with chopped parsley.
\end{directions}

\end{recipe}
