\begin{recipe}{Lobster Dip}{Angela Mangano}{}

\begin{ingredients}
\item 2~celery stalks
\item \oz{8} \theme{cream cheese}, softened
\item lemon juice to taste
\item \half~onion
\item 3-4 boiled \theme{lobster} tails
\item dash of Tabasco sauce
\end{ingredients}

\begin{directions}
\item Chop celery and onion finely.
\item Chop lobster.
\item Combine all ingredients.
\item Chill.
\end{directions}

\end{recipe}
