\begin{recipe}{Lemon Squares}{Lydia Rowland}{16~squares}

\begin{ingredients}
\item \C{\half} butter, at room-temperature
\item \C{\quarter} confectioner's sugar
\item \C{1} all-purpose flour
\item \tp{\half} salt
\item 2~eggs, beaten
\item \C{1} sugar
\item \Tp{3} flour
\item zest of 1~lemon
\item juice of 1~lemon
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{350}.
\item Beat together butter, confectioner's sugar, \C{1} flour, and salt.
\item Pat into a greased \inch{8} square pan.
\item Bake until pastry is a creamy gold. Approximately 20~minutes.
\item Beat together eggs, sugar, \Tp{3} flour, lemon juice, and lemon zest.
\item Pour over crust.
\item Bake until set. Approximately 15 to 20~minutes.
\item Dust with confectioner's sugar.
\end{directions}

\end{recipe}
