\begin{recipe}{June's Brownies}{}{}

\begin{ingredients}
\item \C{\half} \theme{nuts}
\item \C{\half} butter
\item \C{\half} flour
\item 1~egg
\item \tp{1} vanilla
\item pinch of salt
\item \Tp{3} \theme{cocoa}
\end{ingredients}

\begin{directions}
\item Preheat oven to \tF{300}.
\item Blend all ingredients together.
\item Pour into an \inch{8} square pan.
\item Bake for 20 to 30~minutes.
\end{directions}

\subrecipe{Icing}
\begin{ingredients}
\item \C{2} confectioner's sugar
\item \C{\half} butter
\item \Tp{2} cocoa
\item \C{\quarter} hot water
\end{ingredients}

\begin{directions}
\item Combine all ingredients, adjust water to control consistency.
\end{directions}

\end{recipe}
