\begin{recipe}{Chocolate Peanut Crispies}{Rosa Gemmel}{}

\begin{ingredients}
\item \C{1} \theme{chocolate} chips
\item \C{1} \theme{peanut butter} chips
\item \Tp{2} oil
\item \C{3\half} corn flakes \theme{cereal}
\item \half{} to \C{1} \theme{peanut}s
\end{ingredients}

\begin{directions}
\item Melt chips with oil in the microwave for 3~minutes.
\item Cool slightly.
\item Stir in corn flakes and peanuts.
\item Drop spoonfuls onto a greased baking sheet.
\item Chill for 30~minutes.
\end{directions}

\end{recipe}
