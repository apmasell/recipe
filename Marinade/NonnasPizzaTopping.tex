\begin{recipe}{Nonna's Pizza Topping}{Nonna}{}

\begin{ingredients}
\item 3~plum tomatoes, seeded, preferably peeled, and chopped
\item 1~stalk young \theme{celery}, diced
\item 1~clove garlic, minced
\item 2~\theme{zucchini}, sliced length-wise
\item salt
\item oregano
\item olive oil
\item \htheme{Parmesan}{cheese}, grated 
\end{ingredients}

\begin{directions}
\item Combine all ingredients.
\item Let stand for at least 30~minutes.
\item When dressing pizza, discard any liquid left behind.
\end{directions}

\end{recipe}
