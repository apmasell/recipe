\begin{recipe}{Green Peppers and Tomatoes}{Nonna}{}

\begin{ingredients}
\item green peppers, sliced
\item tomato pieces
\item olive oil
\item egg, beaten
\end{ingredients}

\begin{directions}
\item Fry tomatoes and peppers in olive oil until peppers are tender.
\item Add egg and stir to distribute.
\end{directions}

\end{recipe}
