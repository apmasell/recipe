\begin{recipe}{Rotisserie Chicken}{}{}

\begin{ingredients}
\item whole \theme{chicken} or \theme{turkey}
\item salt
\item pepper
\item smoked paprika
\item olive oil
\end{ingredients}

\begin{directions}
\item Coat chicken with oil lightly.
\item Season with spices and rub them in.
\item Put on a spit.
\item Add smoking chips to the barbecue.
\item Cover grill with a drip pan.
\item Cook over medium heat. Approximately 1~hour for a chicken. Consult \seerecipe{Turkey:OpenPanRoastTurkey} for times.
\item For last \half~hour, baste with drippings.
\end{directions}

\hint{Balancing the rotisserie is an art.}
\end{recipe}
