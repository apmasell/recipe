\begin{recipe}{Pineapple Salsa Chicken}{the Steffler Family}{}

\begin{ingredients}
\item 1~can crushed \theme{pineapple}
\item 4~\theme{chicken} breasts
\item 1~clove pressed garlic
\item \tp{1} cumin
\item \Tp{1} oil
\item \C{\half} red \theme{bell pepper}, minced
\item \C{\quarter} green bell pepper, minced
\item \Tp{1} green onion, minced
\item \tp{2} cilantro, minced
\item \tp{2} jalapeño peppers, minced
\item \tp{1} lime zest
\item salt and pepper
\end{ingredients}

\begin{directions}
\item Rub chicken with garlic.
\item Sprinkle with cumin, slat and pepper.
\item Sauté chicken in hot oil over medium-high heat until browned, turning once.
\item Add \C{\half} pineapple juice to chicken.
\item Reduce heat.
\item Simmer for 7 to 10~minutes.
\item In a bowl, combine pineapple, remaining juice, peppers, cilantro, onions and lime.
\item Cut each breast into slices and serve with salsa.
\end{directions}
\end{recipe}
