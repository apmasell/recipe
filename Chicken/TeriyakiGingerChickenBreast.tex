\begin{recipe}{Teriyaki Ginger Chicken Breast}{Lydia Rowland}{}

\begin{ingredients}
\item \lbs{1} \theme{chicken} breast
\item \Tp{3} soya sauce
\item \Tp{1} brown sugar
\item \Tp{1} dry \theme{sherry}
\item \tp{\half} \theme{gingerroot}, grated 
\item 1~minced clove of garlic
\end{ingredients}

\begin{directions}
\item Mix all ingredients.
\item Let marinate for 1~hour.
\item Cook chicken.
\end{directions}

\end{recipe}
