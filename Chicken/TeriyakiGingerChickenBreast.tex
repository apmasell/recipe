\begin{recipe}{Teriyaki Ginger Chicken Breast}{Lydia Rowland}{}

\begin{ingredients}
\item \lbs{1} chicken breast
\item \Tp{3} soya sauce
\item \Tp{1} brown sugar
\item \Tp{1} dry sherry
\item \tp{\half} grated fresh ginger
\item 1~minced clove of garlic
\end{ingredients}

\begin{directions}
\item Mix all ingredients.
\item Let marinate for 1~hour.
\item Cook chicken.
\end{directions}

\end{recipe}
