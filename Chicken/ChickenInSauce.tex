\begin{recipe}{Chicken in Sauce}{Nonna}{}

\begin{ingredients}
\item \theme{chicken} pieces
\item tomato pieces
\item tomato purée
\item garlic
\item basil
\item olive oil
\item salt
\end{ingredients}

\begin{directions}
\item Fry chicken pieces in olive oil.
\item Remove chicken.
\item Fry garlic and tomato pieces.
\item Add chicken, basil, and purée.
\item Salt to taste.
\item Cook gently for as long as possible.
\item Let stand for at least 1~hour.
\end{directions}

\hint{Remaining sauce can be used over pasta.}
\end{recipe}
