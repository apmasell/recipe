\begin{recipe}{Wax}{}{}

\begin{ingredients}
\item 1~non-toxic crayon
\item \gr{140} paraffin wax
\item \gr{70} mineral oil
\end{ingredients}

\begin{directions}
\item Combine all ingredients in a small pot.
\item Heat over \emph{low} heat, stirring regularly.
\item Once melted, dip cheese in wax to coat.
\end{directions}

\hint{Use the lowest possible heat. Wax heats up much more quickly than water.}
\hint{Wax maybe allowed to solidify and the be melted again.}
\hint{A pot should be dedicated to this task.}
\end{recipe}
