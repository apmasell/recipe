\begin{recipe}{Hard Cheese}{David Fankhauser}{\lbs{1}}

\begin{ingredients}
\item \ltr{4} whole milk
\item \C{\half} milk powder
\item \half{}~tablet rennet
\item \C{\quarter} \theme{buttermilk}, or buttermilk \seerecipe{FrozenStock}
\item 1 to \tp{2} salt
\end{ingredients}

\subrecipe{Evening Day 1}
\begin{directions}
\item Mix milk and powder in a large, sanitised double boiler.
\item Warm milk to \tC{20}.
\item Inoculate the milk.
\item Cover with a sanitised lid.
\end{directions}

\subrecipe{Morning Day 2}
\begin{directions}
\item Warm the milk to \tC{30}.
\item Dissolve rennet in \C{\quarter} cold water.
\item Stir rennet into milk.
\item Let sit undisturbed for 1 to 2~hours until a clean break is achieved.
\item Cut the curd.
\item Cook the curd, to \tC{34} for a soft cheese up to \tC{39} for a firm cheese, for 15~minutes while gently lifting curd from the bottom of the pot.
\item Remove from heat.
\item Let stand 10~minutes.
\item Filter whey through a cloth.
\item Add salt to the curd and mix in thoroughly.
\item Place cheese in a cloth and in a cheese press.
\item Press at room temperature for 12~hours.
\end{directions}

\subrecipe{Evening Day 2}
\begin{directions}
\item Remove cheese from the press.
\item Rub with salt.
\item Wrap in a fresh cloth.
\item Place in the refrigerator on a rack.
\end{directions}

\subrecipe{Daily until Rind Forms}
\begin{directions}
\item Replace cloth if wet.
\end{directions}

\subrecipe{After Rind Forms}
\begin{directions}
\item Coat with \seerecipe{Cheese:Wax}.
\item Let age in the refrigerator for at least 1~month.
\end{directions}

\end{recipe}
