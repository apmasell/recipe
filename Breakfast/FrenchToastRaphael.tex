\begin{recipe}{French Toast Raphael}{the Steffler Family}{}

\begin{ingredients}
\item \C{6} white bread
\item \oz{6} cubed \theme{cream cheese}
\item 6 well beaten \theme{egg}s
\item \C{1} milk
\item \tp{\half} cinnamon
\item \C{\third} maple syrup
\end{ingredients}

\begin{directions}
\item Remove crusts of bread and cut into \cm{2} cubes.
\item Place half the bread in a greased \inch{8} baking pan.
\item Dot cheese on top.
\item Cover with remaing bread.
\item Combine remaning ingredients and pour in pan.
\item Cover with plastic wrap and refridgerate over night.
\item Preheat oven to \tF{375}.
\item Bake for 45~minutes.
\end{directions}
\end{recipe}
