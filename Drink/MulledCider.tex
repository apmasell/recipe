\begin{recipe}{Mulled Cider\FIXME}{}{}

\begin{ingredients}
\item \ltr{1} cranberry juice
\item \ltr{2} \theme{apple} juice
\item 6~cloves
\item 2~cinnamon sticks
\end{ingredients}

\begin{directions}
\item Combine all ingredients in a pot.
\item Simmer 15 to 20~minutes.
\end{directions}

\hint{If desired, add rum before serving.}
\end{recipe}
