\begin{recipe}{Irish Cream 2}{}{}

\begin{ingredients}
\item \mL{395} Carnation-brand condensed milk
\item \mL{300} Eagle-brand condensed milk
\item \tp{1\half} chocolate drink mix
\item \Tp{1} instant coffee
\item 2~eggs
\item \C{1\half} whiskey
\end{ingredients}

\begin{directions}
\item Blend all ingredients until smooth.
\item Refrigerate for 24~hours.
\end{directions}

\end{recipe}
